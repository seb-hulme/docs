---
title: "List Contacts"
api: "GET /api/contacts"
description: "Retrieve a list of contacts in your workspace"
---

Fetch all contacts who have replied to your campaigns.

## Query Parameters

<ParamField query="client_id" type="string">
  Filter contacts by client UUID
</ParamField>

<ParamField query="search" type="string">
  Search by email address, name, or company
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of contacts to return (max 100)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of contacts to skip for pagination
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of contact objects
  <Expandable title="Contact Object">
    <ResponseField name="id" type="string">
      Unique contact identifier (UUID)
    </ResponseField>
    <ResponseField name="email" type="string">
      Contact's email address
    </ResponseField>
    <ResponseField name="first_name" type="string">
      Contact's first name (if available)
    </ResponseField>
    <ResponseField name="last_name" type="string">
      Contact's last name (if available)
    </ResponseField>
    <ResponseField name="company" type="string">
      Contact's company name (if available)
    </ResponseField>
    <ResponseField name="title" type="string">
      Contact's job title (if available)
    </ResponseField>
    <ResponseField name="reply_count" type="integer">
      Number of replies from this contact
    </ResponseField>
    <ResponseField name="last_reply_at" type="string">
      ISO timestamp of their most recent reply
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO timestamp when contact was created
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata
</ResponseField>

## Examples

<CodeGroup>
```bash cURL
curl -X GET "https://app.replyify.ai/api/contacts?limit=20" \
  -H "Authorization: Bearer rpl_your_api_key"
```

```javascript JavaScript
const response = await fetch(
  'https://app.replyify.ai/api/contacts?limit=20',
  {
    headers: {
      'Authorization': 'Bearer rpl_your_api_key'
    }
  }
);
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.replyify.ai/api/contacts',
    params={'limit': 20},
    headers={'Authorization': 'Bearer rpl_your_api_key'}
)
data = response.json()
```
</CodeGroup>
